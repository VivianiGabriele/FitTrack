function createModernChart(canvasId, measurementData, goalData, labels, color) { const ctx = document.getElementById(canvasId); if (!ctx) return; const container = ctx.parentElement; ctx.width = container.offsetWidth; ctx.height = container.offsetHeight; const isFatChart = canvasId === 'fatChart'; const isWeightChart = canvasId === 'weightChart'; const unit = isWeightChart ? 'kg' : '%'; const label = isWeightChart ? 'Weight' : 'Body Fat'; // Creazione gradiente const gradient = ctx.getContext('2d').createLinearGradient(0, 0, 0, ctx.height); gradient.addColorStop(0, color + '80'); gradient.addColorStop(1, color + '10'); // Formattazione date (assicurati che labels sia un array di stringhe 'YYYY-MM-DD') const formattedLabels = labels.map(dateStr => { if (!dateStr) return ''; const [year, month, day] = dateStr.split('-'); return `${day}/${month}/${year.slice(-2)}`; }); // Calcolo massimo valore per l'asse Y (body fat) const maxValue = isFatChart ? Math.max(...measurementData.filter(v => v !== null), ...goalData.filter(v => v !== null)) * 1.2 : undefined; return new Chart(ctx, { type: 'line', data: { labels: formattedLabels, datasets: [ { label: 'Measurements', data: measurementData, borderColor: color, backgroundColor: gradient, borderWidth: 3, tension: 0.4, fill: true, pointBackgroundColor: color, pointRadius: 4, pointHoverRadius: 6, pointBorderColor: '#fff', pointBorderWidth: 2 }, { label: 'Goals', data: goalData, borderColor: '#4CAF50', borderWidth: 2, borderDash: [5, 3], tension: 0, fill: false, pointBackgroundColor: '#4CAF50', pointRadius: 4, pointHoverRadius: 6, pointStyle: 'rect' } ] }, options: { responsive: true, maintainAspectRatio: false, plugins: { legend: { position: 'top', labels: { color: '#fff', usePointStyle: true, padding: 20 } }, tooltip: { mode: 'index', intersect: false, backgroundColor: 'rgba(0,0,0,0.7)', titleColor: '#fff', bodyColor: '#fff', borderColor: 'rgba(255,255,255,0.1)', borderWidth: 1 } }, scales: { y: { beginAtZero: isFatChart, max: isFatChart ? maxValue : undefined, ticks: { color: 'rgba(255,255,255,0.7)', stepSize: isFatChart ? 5 : undefined, callback: isFatChart ? (value) => value + '%' : undefined }, grid: { color: 'rgba(255,255,255,0.1)' } }, x: { ticks: { color: 'rgba(255,255,255,0.7)', maxRotation: 0, minRotation: 0 }, grid: { display: false } } } } }); }
